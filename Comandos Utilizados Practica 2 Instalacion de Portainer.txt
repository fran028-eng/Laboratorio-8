Comandos Utilizados Practica 2 Instalacion de Portainer 

---Instalar y ejecutar Portainer---


# 1. Descargar imagen Portainer CE
docker pull portainer/portainer-ce:latest

# 2. Crear volumen persistente
docker volume create portainer_data

# 3. Ejecutar Portainer
docker run -d \
  -p 9443:9443 \
  -p 8000:8000 \
  --name portainer \
  --restart=always \
  -v /var/run/docker.sock:/var/run/docker.sock \
  -v portainer_data:/data \
  portainer/portainer-ce:latest

# 4. Verificar contenedor
docker ps

# 5. (Opcional) Revisar logs
docker logs -f portainer
