Comandos Utilizados Practica 2 Instalacion de Portainer 


#Descargar la imagen de Portainer

docker pull portainer/portainer-ce

#Verificamos

docker images

#Crear y ejecutar el contenedor de Portainer

docker volume create portainer_data

#Lo ejecutamos

docker run -d \
  -p 9443:9443 \
  -p 8000:8000 \
  --name portainer \
  --restart=always \
  -v /var/run/docker.sock:/var/run/docker.sock \
  -v portainer_data:/data \
  portainer/portainer-ce
